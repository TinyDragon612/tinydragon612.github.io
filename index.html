TrieSpeller

A project for CMSC 14100 using tries to make a spell-checker.

class Trie:

    def __init__(self, root):
        '''
        Constructor

        Parameters:
         root : str (a character, but Python does not have a "char" type)

        Initialize root to given char, empty dict of children, final to False.
        '''
        self.root = root
        self.children = {}
        self.final = False

    def add(self, word):
        '''
        Add the word to the trie.

        Parameters:
	     word : str

        Returns: (does not return a value)
        '''
        node = self
        for letter in word:
            if letter not in node.children:
                node.children[letter] = Trie(letter)
            node = node.children[letter]
        node.final = True

    def is_word(self, word):
        '''
        Check presence of given word in the trie.

        Parameters:
         word : str

        Returns: boolean
        '''
        node = self
        for letter in word:
            if letter not in node.children:
                return False
            node = node.children[letter]
        return node.final

    def list_words(self):
        '''
        Return all the words in the trie. Returned list not guaranteed
        in any particular order.

        Parameters:
         none

        Returns: list[str]
        '''
        words = []
        def recur(node, prefix):
            if node.final:
                words.append(prefix)
            for letter, child in node.children.items():
                recur(child, prefix + letter)

        recur(self, "")
        return words

    def num_words(self):
        '''
        Return the number of words in the trie.

        Parameters:
          none

        Returns: int
        '''
        return len(self.list_words())

    def complete(self, prefix):
        '''
        Return all completions given prefix. The returned list is not
        guaranteed to be in any particular order.

        Parameters:
          prefix : str

        Returns: list[str]
        '''
        return self._compl(prefix, "")

    def _compl(self, prefix, acc):
        '''
        Private method. Return all completions given prefix. The
        parameter acc stores the string seen thus far in traversal of
        the trie. The returned list is not guaranteed to be in any
        particular order.

        Parameters:
          prefix : str
          acc : str

        Returns: list[str]
        '''
        if prefix:
            start = prefix[0]
            if start not in self.children:
                return []
            return self.children[start]._compl(prefix[1:], acc + start)
        words = []
        if self.final:
            words.append(acc)
        for letter, child in self.children.items():
            words.extend(child._compl("", acc + letter))
        return words

    def num_starting(self, prefix):
        '''
        Return the number of completions of the given prefix.

        Parameters:
          prefix : str

        Returns: int
        '''
        return len(self.complete(prefix))

    def num_next(self, prefix):
        '''
        Return the number of next letters given prefix.

        Parameters:
          prefix : str

        Returns: int
        '''
        node = self
        for letter in prefix:
            if letter not in node.children:
                return 0
            node = node.children[letter]
        return len(node.children)

class TrieSpeller:
    """
    Class for a trie-based speller
    """

    def __init__(self):
        '''
        Constructor

        Parameters:
          none

        Initialize dictionary of empty tries, one per letter.
        '''
        self.tries = {}
        for char in 'abcdefghijklmnopqrstuvwxyz':
            self.tries[char] = Trie(char)

    def add(self, word):
        '''
        Add the word to the trie if it consists only of lowercase
        letters.

        Parameters:
          word : str

        Returns: (does not return a value)
        '''
        if word == word.lower() and word.isalpha():
            self.tries[word[0]].add(word[1:])

    def add_from_file(self, filename):
        '''
        Read the named file, add words (one per line in file).

        Parameters:
          filename : str

        Returns: (does not return a value)
        '''
        try:
            with open(filename, 'r') as f:
                for line in f:
                    self.add(line.strip())
        except FileNotFoundError:
            return None

    def is_word(self, word):
        '''
        Check presence of given word in the object.

        Parameters:
          word : str

        Returns: boolean
        '''
        start = word[0]
        node = self.tries[start]
        for letter in word[1:]:
            if letter not in node.children:
                return False
            node = node.children[letter]
        return node.final

    def complete(self, prefix):
        '''
        Return all completions given prefix. The returned list is not
        guaranteed to be in any particular order.

        Parameters:
          prefix : str

        Returns: list[str]
        '''
        start = prefix[0]
        words = self.tries[start].complete(prefix[1:])
        return set([start + w for w in words])

    def num_starting(self, prefix):
        '''
        Return the number of completions given prefix.

        Parameters:
          prefix : str

        Returns: int
        '''
        return len(self.complete(prefix))

    def num_next(self, prefix):
        '''
        Return the number of next letters given prefix.

        Parameters:
          prefix : str

        Returns: int
        '''
        node = self.tries[prefix[0]]
        for letter in prefix[1:]:
            if letter not in node.children:
                return 0
            node = node.children[letter]
        return len(node.children)

    def list_words(self):
        '''
        Return all the words in the object. Returned list not
        guaranteed in any particular order.

        Parameters:
          none

        Returns: list[str]
        '''
        words = []
        for char, trie in self.tries.items():
            words.extend([char + w for w in trie.list_words()])
        return set(words)

    def num_words(self):
        '''
        Return the number of words in the object.

        Parameters:
          none

        Returns: int
        '''
        return len(self.list_words())
